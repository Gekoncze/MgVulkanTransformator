        public Array(int... values){
            this(values.length);
            for(int i = 0; i < values.length; i++) get(i).setValue(values[i]);
        }

        public Array(boolean... values){
            this(values.length);
            for(int i = 0; i < values.length; i++) get(i).setValue(values[i] == false ? Vk.VK_FALSE : Vk.VK_TRUE);
        }
